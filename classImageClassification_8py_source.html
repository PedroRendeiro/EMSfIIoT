96  ksize=-1)

98  (minVal, maxVal) = (np.min(gradX), np.max(gradX))
99  gradX = (255 * ((gradX - minVal) / (maxVal - minVal)))
100  gradX = gradX.astype("uint8")
101 
102  # apply a closing operation using the rectangular kernel to help
103  # cloes gaps in between credit card number digits, then apply
104  # Otsu's thresholding method to binarize the image
105  gradX = cv2.morphologyEx(tophat, cv2.MORPH_CLOSE, self.rectKernel)
106  thresh = cv2.threshold(gradX, 0, 255,
