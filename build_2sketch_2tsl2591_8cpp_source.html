159 float TSL2591::calculateLux(uint16_t ch0, uint16_t ch1)
160 {
161  float atime, again;
162  float cpl, lux1, lux2, lux;
163  uint32_t chan0, chan1;
164 
165  // Check for overflow conditions first
166  if ((ch0 == 0xFFFF) | (ch1 == 0xFFFF))
167  {
168  // Signal an overflow
169  return -1;
170  }
171 
172  // Note: This algorithm is based on preliminary coefficients
173  // provided by AMS and may need to be updated in the future
174 
175  switch (_integration)
176  {
177  case TSL2591_INTEGRATIONTIME_100MS :
178  atime = 100.0F;
179  break;
180  case TSL2591_INTEGRATIONTIME_200MS :
181  atime = 200.0F;
182  break;
183  case TSL2591_INTEGRATIONTIME_300MS :
184  atime = 300.0F;
185  break;
186  case TSL2591_INTEGRATIONTIME_400MS :
187  atime = 400.0F;
188  break;
189  case TSL2591_INTEGRATIONTIME_500MS :
190  atime = 500.0F;
191  break;
192  case TSL2591_INTEGRATIONTIME_600MS :
193  atime = 600.0F;
194  break;
195  default: // 100ms
196  atime = 100.0F;
197  break;
198  }
199 
200  switch (_gain)
201  {
202  case TSL2591_GAIN_LOW :
203  again = 1.0F;
204  break;
205  case TSL2591_GAIN_MED :
206  again = 25.0F;
207  break;
208  case TSL2591_GAIN_HIGH :
209  again = 428.0F;
210  break;
211  case TSL2591_GAIN_MAX :
212  again = 9876.0F;
213  break;
214  default: // MED
215  again = 25.0F;
216  break;
217  }
218 
219  cpl = (atime * again) / TSL2591_LUX_DF;
220 
221  lux = ( ((float)ch0 - (float)ch1 )) * (1.0F - ((float)ch1/(float)ch0) ) / cpl;
222 
223  // Return lux value
224  return lux;
225 }
